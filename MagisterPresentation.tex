\section{Постановка задачи}



\begin{frame}{Постановка задачи}
	\quad\quad Рассмотрим уравнение равновесия сплошной среды
	\begin{gather}
		\label{EquilibriumEquation}
		\nabla \cdot \boldsymbol{\widehat{\sigma}} + \boldsymbol{b} = \boldsymbol{0},
	\end{gather}
	где тензор напряжений $\boldsymbol{\widehat{\sigma}}$ определяется следующим образом
	\begin{gather}
		\label{DefineSigma}
		\boldsymbol{\widehat{\sigma}} =
		p_1 \widehat{\text{\textbf{C}}} \cdot \cdot \boldsymbol{\widehat{\varepsilon}} +
		p_2 \iint\limits_{S'(\boldsymbol{x}) \cap S}
		\varphi(\boldsymbol{x}, \boldsymbol{x}') \widehat{\text{\textbf{C}}} \cdot \cdot \boldsymbol{\widehat{\varepsilon}} dS'(\boldsymbol{x}),
		\quad
		\boldsymbol{x'} \in S'(\boldsymbol{x}),
	\end{gather}
	где $p_1 + p_2 = 1$, причём $p_1 > 0$ и $p_2 \geqslant 0$; 
	
	$\varphi$ --- некоторая нормированная положительная функция в области $S'(\boldsymbol{x})$;
	
	$S'(\boldsymbol{x})$ --- область нелокального влияния.
	
	\quad\quad Будем рассматривать кинематические и силовые граничные условия
	\begin{gather*}
		\boldsymbol{u}|_{\Gamma_1} = \boldsymbol{d}(\boldsymbol{x}), 
		\quad
		\boldsymbol{\widehat{\sigma}} \cdot \boldsymbol{n}|_{\Gamma_2} = \boldsymbol{p}(\boldsymbol{x}),
	\end{gather*}
	где $\Gamma_1, \Gamma_2 \subset \partial S$ и $\Gamma_1 \cap \Gamma_2 = \emptyset$.
\end{frame}



\begin{frame}
	\quad\quad Положим, что перемещения малы, поэтому для определения компонент тензора деформаций $\boldsymbol{\widehat{\varepsilon}}$ воспользуемся соотношением Коши
	\begin{gather*}
		\label{DefineStrains}
		\varepsilon_{ij} = \dfrac{1}{2} \left( u_{i, j} + u_{j, i} \right), \quad i, j = \overline{1, 2},
	\end{gather*}
	где $u_i$, $i = \overline{1, 2}$ --- компоненты вектора перемещения $\boldsymbol{u}$.
	
	\quad\quad Считаем, что рассматриваемый случай соответствует случаю линейного упругого изотропного тела для задачи в плоском напряжённом состоянии, поэтому компоненты тензора коэффициентов упругости $\widehat{\text{\textbf{C}}}$ определим следующим образом:
	\begin{gather*}
		C_{ijkl} =
		\dfrac{\nu E}{1 - \nu^2} \delta_{ij} \delta_{kl}
		+
		\dfrac{E}{2(1+\nu)} (\delta_{ik} \delta_{jl} + \delta_{il} \delta_{jk}), \quad
		i, j, k, l = \overline{1, 2},
	\end{gather*}
	где $E$ --- модуль Юнга; $\nu$ --- коэффициент Пуассона.

	\quad\quad В расчётах будем использовать функцию нелокального влияния
	\begin{gather*}
		\varphi(|\boldsymbol{x} - \boldsymbol{x}'|) =
		\begin{cases}
			\dfrac{1}{\pi r^2} \left( 1 - \dfrac{|\boldsymbol{x} - \boldsymbol{x}'|^2}{r^2} \right), \quad &|\boldsymbol{x} - \boldsymbol{x}'| \leqslant r, \\
			0, &|\boldsymbol{x} - \boldsymbol{x}'| > r,
		\end{cases}
	\end{gather*}
	где $r$ --- радиус нелокального влияния.
\end{frame}



\section{Численный алгоритм решения}



\begin{frame}{Численный алгоритм решения}
	\quad\quad К уравнению (\ref{EquilibriumEquation}) применим стандартную конечно-элементную процедуру аппроксимации и подставим в полученное уравнение (\ref{DefineSigma}). В итоге, перейдя к индексной форме записи, получим
	\begin{multline*}
		p_1 \iint\limits_S N_{n, i}^{(e)} C_{ijkl} \varepsilon_{kl} dS
		+
		p_2 \iint\limits_S
		N_{n, i}^{(e)}
		\iint\limits_{S'(\boldsymbol{x}) \cap S}
		\varphi(\boldsymbol{x}, \boldsymbol{x}') C_{ijkl} \varepsilon_{kl}
		dS'(\boldsymbol{x})
		dS 
		= \\ =
		\oint\limits_{\Gamma_2} N_n^{(e)} p_j d\Gamma
		-
		\iint\limits_S N_n^{(e)} b_j dS, \quad
		i, j, k, l = \overline{1, 2}, \ n \in I^{(e)}, \ (e) \in S_h.
	\end{multline*}
	
	\quad\quad Итоговое выражение, которое мы хотим получить, выглядит следующим образом
	\begin{gather*}
		\left( p_1 \widehat{\text{\textbf{K}}}^{Loc} + p_2 \widehat{\text{\textbf{K}}}^{NonLoc} \right) \cdot \cdot \widehat{\text{\textbf{U}}} = \widehat{\text{\textbf{P}}} - \widehat{\text{\textbf{B}}}.
	\end{gather*}
	
	\quad\quad Для простоты дальнейшего изложения введём тензор $\widehat{\text{\textbf{K}}}_{nm}^{(e)(e')}$, который по сути представляет из себя блок $2 \times 2$ матрицы элемента $(e)$. Коэффициенты блока имеют вид
	\begin{gather*}
		K_{nmij}^{(e)(e')} (\boldsymbol{x}, \boldsymbol{y}) = 
		\delta_{np} \delta_{mq} C_{ikjl} 
		N_{n, k}^{(e)}(\boldsymbol{x}) 
		N_{m, l}^{(e')} (\boldsymbol{y}) \boldsymbol{e}_{i} \otimes \boldsymbol{e}_j \otimes 	\boldsymbol{E}_p \otimes \boldsymbol{E}_q, \\
		i,j,k,l = \overline{1, 2},
		\quad
		n,m,p,q = \overline{1, D}.
	\end{gather*}
\end{frame}



\begin{frame}
	\quad\quad Аппроксимируем локальную матрицу жёсткости и векторы плотности поверхностных и объёмных сил
	\begin{gather}
		\label{KLocApprox1}
		\widehat{\text{\textbf{K}}}^{Loc} =
		\sum\limits_{(e) \in S_h}
		\sum\limits_{n \in I^{(e)}}
		\sum\limits_{m \in I^{(e)}}
		\sum\limits_{q \in Q^{(e)}}
		w_q \widehat{\text{\textbf{K}}}_{nm}^{(e)(e)}(\boldsymbol{x}_q, \boldsymbol{x}_q) J_q^{(e)},
	\end{gather}	
	\begin{gather*}	
		\widehat{\text{\textbf{P}}} =
		\sum\limits_{(be) \in \Gamma_h}
		\sum\limits_{n \in I^{(be)}} 
		\boldsymbol{E}_n \otimes
		\sum\limits_{q \in Q^{(be)}}
		w_q N_n^{(be)}(\boldsymbol{x}_q) \boldsymbol{p}(\boldsymbol{x}_q) J_q^{(be)},
	\end{gather*}	
	\begin{gather*}
		\widehat{\text{\textbf{B}}} =
		\sum\limits_{(e) \in S_h}
		\sum\limits_{n \in I^{(e)}}
		\boldsymbol{E}_n \otimes
		\sum\limits_{q \in Q^{(e)}}
		w_q N_{n}^{(e)} (\boldsymbol{x}_q) \boldsymbol{b} (\boldsymbol{x}_q) J_q^{(e)},
	\end{gather*}
	где $I^{(e)}$ --- множество номеров узлов элементов $(e)$;
	
	$Q^{(e)}$ --- множество номеров квадратурных узлов на элементе $(e)$;
	
	$\boldsymbol{x}_q$ --- коордианата квадратурного узла $q$;

	$w_q$ --- квадратурный вес в квадратурном узле $q$;

	$J_q^{(e)} = \left|\det{\widehat{\text{\textbf{J}}}^{(e)}(\boldsymbol{x}_q)}\right|$ --- якобиан вычисленный в квадратурной точке $\boldsymbol{x}_q$;
	
	$\boldsymbol{E}_n$ --- единичный вектор размерности $D$.
\end{frame}



\begin{frame}
	\quad\quad Аппроксимацию вложенного интеграла организуем относительно квадратурных узлов, что наглядно проиллюстрировано на рис. \ref{fig:ApproxSQ} и приводит нас к следующей формуле
	\begin{multline}
		\label{KNonLocApprox1}
		\widehat{\text{\textbf{K}}}^{NonLoc} =
		\sum\limits_{(e) \in S_h}
		\sum\limits_{n \in I^{(e)}}
		\sum\limits_{q \in Q^{(e)}}
		w_q J_q^{(e)}
		\times \\ \times
		\sum\limits_{(e') \in S_h^{q}}
		\sum\limits_{m' \in I^{(e')}}
		\sum\limits_{q' \in Q^{(e')}}
		w_{q'} \varphi(\boldsymbol{x}_q, \boldsymbol{x}_{q'}) \widehat{\text{\textbf{K}}}_{nm'}^{(e)(e')} (\boldsymbol{x}_q, \boldsymbol{x}_{q'}) J_{q'}^{(e')}.
	\end{multline}	
	
	\begin{figure}[h]
		\centering

		\caption{Аппроксимация зоны нелокального влияния относительно квадратурного узла сетки}
		\label{fig:ApproxSQ}
	\end{figure}
\end{frame}



\begin{frame}
	\quad\quad Можно упростить аппроксимацию вложенного интеграла если проводить её относительно центров элементов. Таким образом радиус поиска следует брать несколько больше, чем радиус нелокального влияния. Тогда расчётная формула (\ref{KNonLocApprox1}) приобретает вид
	\begin{multline}
		\label{KNonLocApprox2}
		\widehat{\text{\textbf{K}}}^{NonLoc} =
		\sum\limits_{(e) \in S_h}
		\sum\limits_{n \in I^{(e)}}
		\sum\limits_{(e') \in S_h^{(e)}}
		\sum\limits_{m' \in I^{(e')}}
		\sum\limits_{q \in Q^{(e)}}
		w_q J_q^{(e)}
		\times \\ \times
		\sum\limits_{q' \in Q^{(e')}}
		w_{q'} \varphi(\boldsymbol{x}_q, \boldsymbol{x}_{q'}) \widehat{\text{\textbf{K}}}_{nm'}^{(e)(e')} (\boldsymbol{x}_q, \boldsymbol{x}_{q'}) J_{q'}^{(e')}.
	\end{multline}
	
	\begin{figure}[h!]
		\centering
		\includegraphics[width=0.35\textwidth]{pics/ApproxS.eps}
		\caption{Аппроксимация зоны нелокального влияния относительно центра конечного элемента}
		\label{fig:ApproxS}
	\end{figure}
\end{frame}



\begin{frame}
	\quad\quad Формулы (\ref{KLocApprox1}) и (\ref{KNonLocApprox2}) не пригодны для параллельных и распределённых вычислений. Чтобы сделать их пригодными для каждого узла $n$ введём множество элементов $E^n$, которым он принадлежит. Тогда аппроксимацию локальной и нелокальной матриц жёсткости можно переписать в следующем виде
	\begin{gather*}
		%\label{KLocApprox3}
		\widehat{\text{\textbf{K}}}^{Loc} =
		\sum\limits_{n \in I^{(e)}}
		\sum\limits_{(e) \in E^{n}}
		\sum\limits_{m \in I^{(e)}}
		\sum\limits_{q \in Q^{(e)}}
		w_q \widehat{\text{\textbf{K}}}_{nm}^{(e)(e)}(\boldsymbol{x}_q, \boldsymbol{x}_q) J_q^{(e)}.
	\end{gather*}
	\begin{multline*}
		%\label{KNonLocApprox3}
		\widehat{\text{\textbf{K}}}^{NonLoc} =
		\sum\limits_{n \in I^{(e)}}
		\sum\limits_{(e) \in E^{n}}
		\sum\limits_{(e') \in S_h^{(e)}}
		\sum\limits_{m' \in I^{(e')}}
		\sum\limits_{q \in Q^{(e)}}
		w_q J_q^{(e)}
		\times \\ \times
		\sum\limits_{q' \in Q^{(e')}}
		w_{q'} \varphi(\boldsymbol{x}_q, \boldsymbol{x}_{q'}) \widehat{\text{\textbf{K}}}_{nm'}^{(e)(e')} (\boldsymbol{x}_q, \boldsymbol{x}_{q'}) J_{q'}^{(e')}.
	\end{multline*}
	Важно отметить, что такой порядок знаков суммирования позволяет собирать матрицу жёсткости построчно, что открывает возможности для эффективного использования многопроцессорных систем.
\end{frame}

	\begin{gather*}
		\boldsymbol{\overline{\sigma}} \cdot \boldsymbol{n} |_{\overline{x}_1 = -1} = -1,

		\boldsymbol{\overline{\sigma}} \cdot \boldsymbol{n} |_{\overline{x}_1 =  1} =  1.

	\begin{gather*}
